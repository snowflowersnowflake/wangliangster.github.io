class Solution(object):
    def plusOne(self, digits):
        """
        :type digits: List[int]
        :rtype: List[int]
        """
        jin = 1
        i = len(digits) - 1
        while i >= 0:
            if jin > 0:
                num = digits[i] + jin
                digits[i] = num % 10
                jin = num / 10
                i = i - 1
            else:
                break
        if i < 0 and jin > 0:
            return [jin] + digits
        else:
            return digits
