```python
class Solution(object):
    def maxSubArray(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        s, minNum, r = 0, 0, None
        for i in range(len(nums)):
            minNum = min(minNum, s)
            s += nums[i]
            r = s - minNum if r is None or s - minNum > r else r
        return r
```
