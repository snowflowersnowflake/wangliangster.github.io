```python
class Solution(object):
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        if x >= 0 and x < 10:
            return True
        if x % 10 == 0 or x < 0:
            return False
        rev_num = 0
        while x > rev_num:
            last_digit = x % 10
            x = x / 10
            if rev_num == x:
                return True
            rev_num = rev_num * 10 + last_digit
            if rev_num == x:
                return True
        return False
```
