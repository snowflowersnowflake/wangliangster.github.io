class Solution(object):
    def readBinaryWatch(self, num):
        """
        :type num: int
        :rtype: List[str]
        """
        def divide(n):
            s = 32
            count = 0
            while n > 0:
                if n >= s:
                    n = n - s
                    count += 1
                s = s / 2
            return count
        def gen(n):
            result = dict()
            for h in range(n):
                count = divide(h)
                if result.has_key(count):
                    result[count].append(h)
                else:
                    result[count] = [h]
            return result
        hours = gen(12)
        minutes = gen(60)
        if num <= 0:
            return ["0:00"]
        result = []
        for i in range(num + 1):
            if hours.has_key(i) and minutes.has_key(num - i):
                for p in hours[i]:
                    for q in minutes[num - i]:
                        result = result + [str(p) + ":" + ("0" + str(q) if q < 10 else str(q))]
        return result
